@mixin hero-text()

Applies hero text styles

Example

scss 
.element {
  @include hero-text;
}
css compiled 
.element h1 {
  font-family: var(--ff-display-condensed);
  text-shadow: var(--shadow-large-title);
}
.element h1::after {
  color: var(--clr-accent-400);
  content: ".";
}
@media screen and (max-width: 23.38em) {
  .element h1::after {
    content: "";
  }
}
.element h1::selection {
  color: var(--clr-secondary-400);
}
.element p {
  text-shadow: var(--shadow-large-body);
}
.element p::selection {
  color: var(--clr-secondary-400);
}
.element .hero__text {
  display: flex;
  flex-direction: column;
  row-gap: 0.75rem;
}
